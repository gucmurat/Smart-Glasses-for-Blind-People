# Smart-Glasses-for-Blind-People

- to create conda environment
  - conda create --name senior-project python=3.9
- activate
  - conda activate senior-project
- deactivate
  - conda deactivate senior-project
- install packages
  - cd ./scripts
  - pip install -r requirements
- running server
  - cd ./scripts/server
  - python server.py
- running client
  - cd ./scripts/client
  - python client.py
